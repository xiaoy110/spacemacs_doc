** f5 切换到init.el
(defun open-init-file()
  (interactive)
  (find-file "~/.emacs.d/init.el"))

;; 这一行代码，将函数 open-init-file 绑定到 <f2> 键上
(global-set-key (kbd "<f5>") 'open-init-file)
;; 移动到行尾
(global-set-key (kbd "M-n") 'move-end-of-line)
